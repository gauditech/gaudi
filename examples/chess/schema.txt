model Player {
    field name string length 2+
}

model Game {
    field black Player
    field white Player
    field winner string nullable one of {"black", "white", "draw"}
    relation moves GameMove
}

model GameMove {
    reference game Game
    field no integer
    field from string regex /[a-h][1-8]/
    field to string regex /[a-h][1-8]/
}

environment py3 {
    engine python3
    # requirements file ./requirements.txt
}
